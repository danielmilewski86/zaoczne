Zadanie #1

Exercise1.run wypisuje na kolorowo tekst z atrybutu Exercise1.colorfulText. Zmodyfikuj ten kod tak, aby uruchomił się od razu po załadowaniu strony i co 5 sekund.

							
var Exercise1 = {
	// nasz przykładowy tekst
	colorfulText: "Litwo! Ojczyzno moja! Ty jesteś jak zdrowie, Ile cię trzeba cenić, ten tylko się dowie, Kto cię stracił. Dziś piękność twą w całej ozdobie Widzę i opisuję, bo tęsknię po tobie Panno święta, co Jasnej bronisz Częstochowy I w Ostrej świecisz Bramie! Ty, co gród zamkowy Nowogródzki ochraniasz z jego wiernym ludem! Jak mnie dziecko do zdrowia powróciłaś cudem...",
	// przypiszmy naszą piaskownicę do atrybutu $sandbox żeby mieć ją zawsze pod ręką
	$sandbox: $('#exercise1 .sandbox'),
	// tu dzieje się cała magia
	run: function(){
		// czyścimy zawartość piaskownicy
		this.$sandbox.empty();
		// dla każdej litery w naszym tekście (split zamieni tekst na tablicę)
		this.colorfulText.split("").map(function( element, index ){
			// losujemy kolor
			var color = Exercise1.getColor(),
			// tworzymy nowy element span w którym ją umieszczamy
			$letter = $('').text(element).css({ color: color, opacity: 0 });
			// dodajemy do piaskownicy i z odpowidnym opóźnieniem pokazujemy
			Exercise1.$sandbox.append( $letter.delay( index * 10 ).animate( { opacity: 1 }, 500 ) );
		});
	},
	getColor: function(){
		// losujemy kolor wykorzystując opcjonalny argument funkcji toString który pozwala nam
		// zamienić wartość numeryczną Math.random do innej reprezentacji, w naszym przypadku
		// do postaci szesnastkowej, po czym odcinamy ostatnie 6 znaków (podanie wartości ujemnej
		// znaczy tyle co "zacznij liczyć od końca")
		return '#' + Math.random().toString(16).slice(-6);
	}
}
